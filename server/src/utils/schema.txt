
///////////////////////
// ENUMS
///////////////////////

enum SportType {
  CRICKET
  FOOTBALL
  RUGBY
  TENNIS
  OTHER
}

enum MatchStatus {
  UPCOMING
  LIVE
  FINISHED
}

///////////////////////
// MODELS
///////////////////////

model Sport {
  id        String    @id @default(uuid())
  name      SportType @unique
  matches   Match[]
  createdAt DateTime  @default(now())
}

model Team {
  id        String   @id @default(uuid())
  name      String
  logo      String?
  players   Player[]
  homeMatches Match[] @relation("HomeTeam")
  awayMatches Match[] @relation("AwayTeam")
}

model Player {
  id        String   @id @default(uuid())
  name      String
  teamId    String
  team      Team     @relation(fields: [teamId], references: [id])
}

model Match {
  id          String      @id @default(uuid())
  title       String
  sportId     String
  sport       Sport       @relation(fields: [sportId], references: [id])

  homeTeamId  String
  awayTeamId  String

  homeTeam    Team        @relation("HomeTeam", fields: [homeTeamId], references: [id])
  awayTeam    Team        @relation("AwayTeam", fields: [awayTeamId], references: [id])

  @@check(homeTeamId != awayTeamId)
  status      MatchStatus @default(UPCOMING)
  startTime   DateTime?
  endTime     DateTime?
  events      Event[]
  createdAt   DateTime    @default(now())
}

model Event {
  id        String   @id @default(uuid())
  matchId   String
  match     Match    @relation(fields: [matchId], references: [id])

  minute    Int?     // useful for football/rugby
  over      String?  // useful for cricket (e.g., 10.2)
  type      String   // e.g., "GOAL", "WICKET", "FOUR", "SIX"
  message   String   // commentary text

  createdAt DateTime @default(now())
}
